###########################################################
## CMAKE SETUP
###########################################################

cmake_minimum_required(VERSION 3.2)
project(lru-cache)

###########################################################
## INCLUDES
###########################################################

include_directories(${CMAKE_CURRENT_SOURCE_DIR}/include)

###########################################################
## SOURCES
###########################################################

set(_SOURCES main.cpp)

###########################################################
## TARGETS
###########################################################

add_executable(lru-cache ${_SOURCES})

###########################################################
## COMPILER FLAGS
###########################################################

target_compile_options(lru-cache PUBLIC -Os -O2 -g -std=c++14)
